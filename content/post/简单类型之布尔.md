# 布尔类型

## 从字符串转换

```python
strValue = "true"  
boolValue = strValue.lower() == "true"  
# boolValue is "True"  
strValue = "false"  
boolValue = bool(strValue)  
# boolValue is True  
boolValue = bool("")  
# boolValue is False
```
```java
String str = "true";
boolean boolValue = Boolean.valueOf(str);
//boolValue is true
```
```javascript
String str = "true";
boolean boolValue = Boolean.valueOf(str);
//boolValue is true
```

```cs
var str = "true";
var boolValue = bool.Parse(str);
//boolValue is True
```

```go
var str = "true"
boolValue,err:=strconv.ParseBool(str)
_ = err
//boolValue is true
```

## 逻辑运算

```python
value1 = True
value2 = False
valueNot1 = (not value1)
# valueNot1 is False
valueNot2 = (not value2)
# valueNot2 is True
valueAnd = value1 and value2
# valueAnd is False
valueOr = value1 or value2
# valueOr is True
valueXor = value1 ^ value2
# valueXor is True
```

```java
boolean value1 = true;
boolean value2 = false;
boolean valueNot1 = !value1;
//valueNot1 is false
boolean valueNot2 = !value2;
//valueNot2 is true
boolean valueAnd = value1 && value2;
//valueAnd is false
boolean valueOr = value1 || value2;
//valueOr is true
boolean valueXor = value1 ^ value2;
//valueXor is true
```

```javascript
var value1 = true;
var value2 = false;
var valueNot1 = !value1;
//valueNot1 is false
var valueNot2 = !value2;
//valueNot2 is true
var valueAnd = value1 && value2;
//valueAnd is false
var valueOr = value1 || value2;
//valueOr is true
var valueXor = value1 ^ value2;
//valueXor is 1
```

```cs
var value1 = true;
var value2 = false;
var valueNot1 = !value1;
//valueNot1 is false
var valueNot2 = !value2;
//valueNot2 is true
var valueAnd = value1 && value2;
//valueAnd is false
var valueOr = value1 || value2;
//valueOr is true
var valueXor = value1 ^ value2;
//valueXor is true
```

```go
var value1 = true
var value2 = false
var valueNot1 = !value1
//valueNot1 is false
var valueNot2 = !value2
//valueNot2 is true
var valueAnd = value1 && value2
//valueAnd is false
var valueOr = value1 || value2
//valueOr is true
```

## 转换为字符串
```python
sunIsStar = True
strSun = str(sunIsStar)
# strStar is "True"
earthIsStar = False
strEarth = "%s" % earthIsStar
# strEarth is "False"
```

```java
boolean sunIsStar = true;
String str1 =
Boolean.toString(sunIsStar);
//str1 is "True"
Boolean earthIsStar = false;
String str2 = earthIsStar.toString();
//str2 is "False"
```

```javascript
var sunIsStar = true;
var str1 = sunIsStar.toString();
//str1 is "true"
var earthIsStar = false;
var str2 = earthIsStar.toString();
//str2 is "false"
```

```cs
var sunIsStar = true;
var str = sunIsStar.ToString();
//str is "True"
```

```go
var sunIsStar = true
var str = strconv.FormatBool(sunIsStar)
```
