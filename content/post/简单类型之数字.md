### BigDecimal

from decimal import Decimal

### Converting to a sting 

```python
import locale  
  
exp = 2.718281828  
s1 = str(exp)  
# s1 is "2.718281828"  
  
s2 = "{0:0.3f}".format(exp)  
# s2 is "2.718"  
  
s3 = "{0:0.2E}".format(exp / 100)  
# s3 is "2.72E-02"  
  
locale.setlocale(locale.LC_ALL, 'ru_Ru')  
s4 = locale.format("%.2f", exp * 1000000, grouping=True)  
# s4 is "2 718 281,83"
```

```java

//s1 is 2.718281828

//s2 is 2,718  
  
String s3 = String.format("%3.2E", exp / 100);  
//s3 is 2,72E-02  
  
String s4 = String.format("%,.2f", exp * 1000_000);  
//"," includes locale-specific grouping characters  
//s4 is 2 718 281,83

DecimalFormatSymbols symbols = formatter.getDecimalFormatSymbols();  
symbols.setGroupingSeparator(' ');  
formatter.setMaximumFractionDigits(2);  
formatter.setDecimalFormatSymbols(symbols);  
String s5 = formatter.format(exp * 1000_000);  
//s5 is 2 718 281,83
```
